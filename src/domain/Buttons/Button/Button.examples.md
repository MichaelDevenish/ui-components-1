#### Buttons with Icons
```jsx
const { FontAwesomeIcon } = require('@Domain/Icons');

<div>
  <div>
    <Button 
      icon={<FontAwesomeIcon type='star' />}
      type='primary-hollow' 
      buttonComponentProps={{onClick:() => alert('primary-hollow')}}
    >
      primary-hollow
    </Button>
    <Button 
      icon={<FontAwesomeIcon type='star' />}
      iconAlignment='right'
      type='primary-hollow' 
      buttonComponentProps={{onClick:() => alert('primary-hollow')}}
    >
      primary-hollow
    </Button>
  </div>
</div>
```
