#### Basic Emoji

```jsx
  <div>
    <Emoji
      emoji='smiley'
      type='xsmall'
    />
    <Emoji
      emoji='smiley'
      type='small'
    />
    <Emoji
      emoji='smiley'
      type='body'
    />
    <Emoji
      emoji='smiley'
      type='heading'
    />
    <Emoji
      emoji='smiley'
      type='display'
    />
    <Emoji
      emoji='smiley'
      type='display-large'
    />
  </div>
```

#### Flag Emoji

```jsx
   <div>
     <Emoji
       emoji='flag-au'
       isFlag={true}
     />
     <Text type='body'>
       Australia Flag
     </Text>
   </div>
```
