```jsx
<div>
  <Brick
    text='Alert!!!'
    color='alert'
  />

  <Brick
    text='Success'
    color='success'
  />

  <Brick
    text='Warning!'
    color='warning'
  />

  <Brick
    text='Primary'
    color='primary'
  />

  <Brick
    text='Neutral (default)'
    color='neutral'
  />

  <Brick
    text='Secondary'
    color='secondary'
  />

  <Brick
    text='Highlight'
    color='highlight'
  />
  <Brick
    text='Dark'
    color='dark'
  />
</div>
```
